# ========== vistas/vista_productos.py ==========
"""
Vista Productos - CRUD de productos
"""
import customtkinter as ctk
import sqlite3
from tkinter import messagebox
from typing import Optional, List

from .componentes.tabla import TablaPersonalizada
from .componentes.formulario import FormularioBase, CampoFormulario
from controladores.producto_controller import ProductoController
from controladores.proveedor_controller import ProveedorController
from modelos.producto import Producto


class VistaProductos(ctk.CTkFrame):
    """Vista CRUD de productos"""

    def __init__(
        self,
        parent,
        conexion: sqlite3.Connection,
        usuario,
        **kwargs
    ):
        super().__init__(parent, fg_color="transparent", **kwargs)

        self.conexion = conexion
        self.usuario = usuario
        self.controller = ProductoController(conexion)
        self.proveedor_controller = ProveedorController(conexion)
        self.producto_seleccionado: Optional[str] = None

        self._crear_widgets()
        self.actualizar()

    def _crear_widgets(self):
        """Crea los widgets de la vista"""
        # Frame superior
        self.frame_superior = ctk.CTkFrame(self, fg_color="transparent")
        self.frame_superior.pack(fill="x", pady=(0, 10))

        # Titulo
        self.lbl_titulo = ctk.CTkLabel(
            self.frame_superior,
            text="Gestion de Productos",
            font=ctk.CTkFont(size=24, weight="bold")
        )
        self.lbl_titulo.pack(side="left")

        # Buscador
        self.frame_busqueda = ctk.CTkFrame(self.frame_superior, fg_color="transparent")
        self.frame_busqueda.pack(side="right")

        self.entry_buscar = ctk.CTkEntry(
            self.frame_busqueda,
            width=250,
            height=35,
            placeholder_text="Buscar por nombre..."
        )
        self.entry_buscar.pack(side="left", padx=(0, 5))
        self.entry_buscar.bind("<KeyRelease>", self._buscar)

        self.btn_buscar = ctk.CTkButton(
            self.frame_busqueda,
            text="Buscar",
            width=80,
            height=35,
            command=self._buscar
        )
        self.btn_buscar.pack(side="left")

        # Frame botones CRUD
        self.frame_botones = ctk.CTkFrame(self, fg_color="transparent")
        self.frame_botones.pack(fill="x", pady=(0, 10))

        self.btn_agregar = ctk.CTkButton(
            self.frame_botones,
            text="+ Agregar",
            width=100,
            height=35,
            fg_color="#28a745",
            hover_color="#218838",
            command=self._abrir_formulario_agregar
        )
        self.btn_agregar.pack(side="left", padx=(0, 5))

        self.btn_editar = ctk.CTkButton(
            self.frame_botones,
            text="Editar",
            width=100,
            height=35,
            fg_color="#007bff",
            hover_color="#0056b3",
            command=self._abrir_formulario_editar
        )
        self.btn_editar.pack(side="left", padx=(0, 5))

        self.btn_eliminar = ctk.CTkButton(
            self.frame_botones,
            text="Eliminar",
            width=100,
            height=35,
            fg_color="#dc3545",
            hover_color="#c82333",
            command=self._eliminar
        )
        self.btn_eliminar.pack(side="left", padx=(0, 5))

        if not self.usuario.es_admin():
            self.btn_eliminar.configure(state="disabled")

        self.btn_actualizar = ctk.CTkButton(
            self.frame_botones,
            text="Actualizar",
            width=100,
            height=35,
            fg_color="#6c757d",
            hover_color="#5a6268",
            command=self.actualizar
        )
        self.btn_actualizar.pack(side="left")

        # Filtro categoria
        self.lbl_filtro = ctk.CTkLabel(
            self.frame_botones,
            text="Categoria:",
            font=ctk.CTkFont(size=12)
        )
        self.lbl_filtro.pack(side="left", padx=(20, 5))

        self.combo_categoria = ctk.CTkComboBox(
            self.frame_botones,
            width=150,
            height=35,
            values=["Todos"],
            command=self._filtrar_categoria
        )
        self.combo_categoria.pack(side="left")
        self.combo_categoria.set("Todos")

        # Tabla
        self.tabla = TablaPersonalizada(
            self,
            columnas=[
                ("id", "ID", 80),
                ("nombre", "Nombre", 180),
                ("precio", "Precio", 100),
                ("categoria", "Categoria", 120),
                ("stock", "Stock", 70),
                ("estado", "Estado", 100)
            ],
            on_select=self._on_seleccionar,
            on_double_click=self._on_doble_clic
        )
        self.tabla.pack(fill="both", expand=True)

    def _buscar(self, event=None):
        """Busca productos por nombre"""
        texto = self.entry_buscar.get().strip()
        if texto:
            productos = self.controller.buscar_por_nombre(texto)
        else:
            productos = self.controller.listar()
        self._cargar_tabla(productos)

    def _filtrar_categoria(self, categoria: str):
        """Filtra productos por categoria"""
        if categoria == "Todos":
            productos = self.controller.listar()
        else:
            productos = self.controller.buscar_por_categoria(categoria)
        self._cargar_tabla(productos)

    def _cargar_tabla(self, productos: List[Producto]):
        """Carga los productos en la tabla"""
        datos = []
        for prod in productos:
            estado = prod.estado_stock()
            datos.append((
                prod.id,
                prod.nombre,
                f"S/ {prod.precio:.2f}",
                prod.categoria or "",
                prod.stock,
                estado
            ))

        def obtener_tag(fila):
            estado = fila[5]
            if estado == "SIN STOCK":
                return "critico"
            elif estado == "STOCK CRITICO":
                return "advertencia"
            return "normal"

        self.tabla.cargar_datos(datos, obtener_tag)

    def _actualizar_categorias(self):
        """Actualiza el combo de categorias"""
        categorias = ["Todos"] + self.controller.obtener_categorias()
        self.combo_categoria.configure(values=categorias)

    def actualizar(self):
        """Actualiza la tabla con todos los productos"""
        productos = self.controller.listar()
        self._cargar_tabla(productos)
        self._actualizar_categorias()
        self.producto_seleccionado = None

    def _on_seleccionar(self, item_id: str):
        """Maneja la seleccion de una fila"""
        self.producto_seleccionado = self.tabla.obtener_id_seleccion()

    def _on_doble_clic(self, item_id: str):
        """Maneja el doble clic para editar"""
        self._abrir_formulario_editar()

    def _obtener_proveedores(self) -> List[str]:
        """Obtiene lista de proveedores para el combo"""
        proveedores = self.proveedor_controller.listar()
        return [f"{p.id} - {p.nombre}" for p in proveedores]

    def _obtener_campos_formulario(self) -> list:
        """Retorna la definicion de campos del formulario"""
        proveedores = [""] + self._obtener_proveedores()
        categorias = [""] + self.controller.obtener_categorias()

        return [
            CampoFormulario("id", "ID", "texto", readonly=True),
            CampoFormulario("nombre", "Nombre", "texto", requerido=True),
            CampoFormulario("precio", "Precio (S/)", "numero", requerido=True),
            CampoFormulario("categoria", "Categoria", "combo", opciones=categorias),
            CampoFormulario("stock", "Stock", "numero", requerido=True, valor_default="0"),
            CampoFormulario("stock_minimo", "Stock Minimo", "numero", valor_default="5"),
            CampoFormulario("proveedor_id", "Proveedor", "combo", opciones=proveedores),
            CampoFormulario("descripcion", "Descripcion", "texto")
        ]

    def _abrir_formulario_agregar(self):
        """Abre el formulario para agregar producto"""
        campos = self._obtener_campos_formulario()
        campos[0].valor_default = "(Automatico)"

        FormularioBase(
            self,
            "Agregar Producto",
            campos,
            self._guardar_producto,
            modo_edicion=False
        )

    def _abrir_formulario_editar(self):
        """Abre el formulario para editar producto"""
        if not self.producto_seleccionado:
            messagebox.showwarning("Atencion", "Seleccione un producto para editar")
            return

        try:
            producto = self.controller.buscar(self.producto_seleccionado)
            datos = producto.to_dict()

            # Formatear proveedor para combo
            if datos.get('proveedor_id'):
                try:
                    prov = self.proveedor_controller.buscar(datos['proveedor_id'])
                    datos['proveedor_id'] = f"{prov.id} - {prov.nombre}"
                except:
                    pass

            FormularioBase(
                self,
                "Editar Producto",
                self._obtener_campos_formulario(),
                self._guardar_producto,
                datos_iniciales=datos,
                modo_edicion=True
            )
        except Exception as e:
            messagebox.showerror("Error", str(e))

    def _guardar_producto(self, datos: dict) -> bool:
        """Guarda o actualiza un producto"""
        try:
            # Extraer proveedor_id del formato combo
            proveedor_id = datos.get('proveedor_id', '')
            if proveedor_id and ' - ' in proveedor_id:
                proveedor_id = proveedor_id.split(' - ')[0]

            if datos.get('id') and datos['id'] != "(Automatico)":
                producto = Producto(
                    id=datos['id'],
                    nombre=datos['nombre'],
                    precio=float(datos['precio']),
                    categoria=datos.get('categoria') or None,
                    stock=int(datos.get('stock', 0)),
                    descripcion=datos.get('descripcion'),
                    stock_minimo=int(datos.get('stock_minimo', 5)),
                    proveedor_id=proveedor_id or None
                )
                self.controller.actualizar(producto)
                messagebox.showinfo("Exito", "Producto actualizado correctamente")
            else:
                producto = Producto(
                    id=None,
                    nombre=datos['nombre'],
                    precio=float(datos['precio']),
                    categoria=datos.get('categoria') or None,
                    stock=int(datos.get('stock', 0)),
                    descripcion=datos.get('descripcion'),
                    stock_minimo=int(datos.get('stock_minimo', 5)),
                    proveedor_id=proveedor_id or None
                )
                self.controller.agregar(producto)
                messagebox.showinfo("Exito", "Producto agregado correctamente")

            self.actualizar()
            return True

        except Exception as e:
            messagebox.showerror("Error", str(e))
            return False

    def _eliminar(self):
        """Elimina el producto seleccionado"""
        if not self.producto_seleccionado:
            messagebox.showwarning("Atencion", "Seleccione un producto para eliminar")
            return

        if not messagebox.askyesno(
            "Confirmar",
            f"Desea eliminar el producto {self.producto_seleccionado}?"
        ):
            return

        try:
            self.controller.eliminar(self.producto_seleccionado)
            messagebox.showinfo("Exito", "Producto eliminado correctamente")
            self.actualizar()
        except Exception as e:
            messagebox.showerror("Error", str(e))

    def _buscar(self, event=None):
        """Busca ventas por ID"""
        texto = self.entry_buscar.get().strip()
        if texto:
            try:
                venta = self.controller.buscar(texto)
                self._cargar_tabla([venta])
            except:
                self._cargar_tabla([])
        else:
            self.actualizar()

    def _filtrar_estado(self, estado: str):
        """Filtra ventas por estado"""
        if estado == "Activas":
            ventas = self.controller.listar_activas()
        elif estado == "Devueltas":
            ventas = [v for v in self.controller.listar() if v.es_devuelta()]
        else:
            ventas = self.controller.listar()
        self._cargar_tabla(ventas)

    def _cargar_tabla(self, ventas: List[Venta]):
        """Carga las ventas en la tabla"""
        datos = []
        total_monto = 0

        for venta in ventas:
            # Obtener nombres
            try:
                cliente = self.cliente_controller.buscar(venta.cliente_id)
                cliente_nombre = cliente.nombre[:20]
            except:
                cliente_nombre = venta.cliente_id

            try:
                producto = self.producto_controller.buscar(venta.producto_id)
                producto_nombre = producto.nombre[:20]
                total = venta.calcular_total_con_precio(producto.precio)
            except:
                producto_nombre = venta.producto_id
                total = 0

            if not venta.es_devuelta():
                total_monto += total

            fecha = venta.fecha[:16] if venta.fecha else ""
            estado = "DEVUELTA" if venta.es_devuelta() else "ACTIVA"

            datos.append((
                venta.id,
                cliente_nombre,
                producto_nombre,
                int(venta.cantidad),
                f"{venta.descuento}%",
                f"S/ {total:.2f}",
                fecha,
                estado
            ))

        def obtener_tag(fila):
            if fila[7] == "DEVUELTA":
                return "devuelta"
            return "normal"

        self.tabla.cargar_datos(datos, obtener_tag)

        # Actualizar resumen
        ventas_activas = len([v for v in ventas if not v.es_devuelta()])
        self.lbl_total_ventas.configure(text=f"Total Ventas Activas: {ventas_activas}")
        self.lbl_monto_total.configure(text=f"Monto Total: S/ {total_monto:,.2f}")

    def actualizar(self):
        """Actualiza la tabla con todas las ventas"""
        ventas = self.controller.listar()
        self._cargar_tabla(ventas)
        self.venta_seleccionada = None

    def _on_seleccionar(self, item_id: str):
        """Maneja la seleccion de una fila"""
        self.venta_seleccionada = self.tabla.obtener_id_seleccion()

    def _on_doble_clic(self, item_id: str):
        """Maneja el doble clic - ver detalles"""
        if self.venta_seleccionada:
            try:
                venta = self.controller.buscar(self.venta_seleccionada)
                total = self.controller.calcular_total_venta(venta.id)

                cliente = self.cliente_controller.buscar(venta.cliente_id)
                producto = self.producto_controller.buscar(venta.producto_id)

                detalle = f"""
                ID: {venta.id}
                Cliente: {cliente.nombre}
                Producto: {producto.nombre}
                Cantidad: {int(venta.cantidad)}
                Precio Unitario: S/ {producto.precio:.2f}
                Descuento: {venta.descuento}%
                Total: S/ {total:.2f}
                Fecha: {venta.fecha}
                Estado: {venta.estado.upper()}
                """
                messagebox.showinfo("Detalle de Venta", detalle)
            except Exception as e:
                messagebox.showerror("Error", str(e))

    def _obtener_clientes(self) -> List[str]:
        """Obtiene lista de clientes para el combo"""
        clientes = self.cliente_controller.listar()
        return [f"{c.id} - {c.nombre}" for c in clientes]

    def _obtener_productos(self) -> List[str]:
        """Obtiene lista de productos con stock para el combo"""
        productos = self.producto_controller.listar()
        return [f"{p.id} - {p.nombre} (Stock: {p.stock})" for p in productos if p.stock > 0]

    def _obtener_campos_formulario(self) -> list:
        """Retorna la definicion de campos del formulario"""
        clientes = self._obtener_clientes()
        productos = self._obtener_productos()

        return [
            CampoFormulario("id", "ID Venta", "texto", readonly=True),
            CampoFormulario("cliente_id", "Cliente", "combo", requerido=True, opciones=clientes),
            CampoFormulario("producto_id", "Producto", "combo", requerido=True, opciones=productos),
            CampoFormulario("cantidad", "Cantidad", "numero", requerido=True, valor_default="1"),
            CampoFormulario("descuento", "Descuento (%)", "slider", rango=(0, 50), valor_default=0)
        ]

    def _abrir_formulario_agregar(self):
        """Abre el formulario para nueva venta"""
        campos = self._obtener_campos_formulario()
        campos[0].valor_default = "(Automatico)"

        FormularioBase(
            self,
            "Nueva Venta",
            campos,
            self._guardar_venta,
            modo_edicion=False
        )

    def _guardar_venta(self, datos: dict) -> bool:
        """Guarda una nueva venta"""
        try:
            # Extraer IDs de los formatos combo
            cliente_id = datos.get('cliente_id', '').split(' - ')[0]
            producto_id = datos.get('producto_id', '').split(' - ')[0]

            venta = Venta(
                id=None,
                cliente=cliente_id,
                producto=producto_id,
                cantidad=float(datos.get('cantidad', 1)),
                descuento=float(datos.get('descuento', 0)),
                fecha=datetime.now().strftime("%Y-%m-%d %H:%M:%S"),
                estado='activa'
            )

            self.controller.agregar(venta, reducir_stock=True)

            # Calcular total para mostrar
            total = self.controller.calcular_total_venta(venta.id)
            messagebox.showinfo(
                "Venta Exitosa",
                f"Venta registrada: {venta.id}\nTotal: S/ {total:.2f}"
            )

            self.actualizar()
            return True

        except Exception as e:
            messagebox.showerror("Error", str(e))
            return False

    def _devolver_venta(self):
        """Marca una venta como devuelta"""
        if not self.venta_seleccionada:
            messagebox.showwarning("Atencion", "Seleccione una venta para devolver")
            return

        try:
            venta = self.controller.buscar(self.venta_seleccionada)
            if venta.es_devuelta():
                messagebox.showwarning("Atencion", "Esta venta ya fue devuelta")
                return

            if not messagebox.askyesno(
                "Confirmar Devolucion",
                f"Desea marcar como devuelta la venta {self.venta_seleccionada}?\n"
                "El stock del producto sera restaurado."
            ):
                return

            self.controller.devolver(self.venta_seleccionada)
            messagebox.showinfo("Exito", "Venta marcada como devuelta")
            self.actualizar()

        except Exception as e:
            messagebox.showerror("Error", str(e))

    def _eliminar(self):
        """Elimina la venta seleccionada"""
        if not self.venta_seleccionada:
            messagebox.showwarning("Atencion", "Seleccione una venta para eliminar")
            return

        if not messagebox.askyesno(
            "Confirmar",
            f"Desea eliminar la venta {self.venta_seleccionada}?\n"
            "Esta accion no se puede deshacer."
        ):
            return

        try:
            self.controller.eliminar(self.venta_seleccionada)
            messagebox.showinfo("Exito", "Venta eliminada correctamente")
            self.actualizar()
        except Exception as e:
            messagebox.showerror("Error", str(e))

    def _generar_factura(self):
        """Genera factura PDF de la venta seleccionada"""
        if not self.venta_seleccionada:
            messagebox.showwarning("Atencion", "Seleccione una venta para generar factura")
            return

        generar_factura_pdf(self.conexion, self.venta_seleccionada)
